country_event = {
	id = sm_smsr.1
	title = sm_smsr.1.t
	desc = sm_smsr.1.d
	picture = DEBATE_REPUBLICAN_eventPicture
	
	trigger = {
		tag = SUM
		OR = {
			government = socialist_state
			government = dictatorship
			government = technocrat_dictatorship
			government = presidental_dictatorship
		}
		has_factions = yes
		OR = {
			faction_influence = {
				faction = mal_reformist_faction
				influence = 25
			}
			faction_influence = {
				faction = mal_democrat_faction
				influence = 25
			}
		}
		is_at_war = no
		faction_in_power = mal_totalitarian_faction
	}
	
	mean_time_to_happen = {
		months = 120
		
		modifier = {
			factor = 0.75
			faction_influence = {
				faction = mal_reformist_faction
				influence = 25
			}
		}
		
		modifier = {
			factor = 0.75
			faction_influence = {
				faction = mal_reformist_faction
				influence = 35
			}
		}
	}
	
	option = {
		name = sm_smsr.1.a
		ai_chance = { factor = 65 }
		add_country_modifier = {
			name = democratic_forces_unpleased
			duration = 548
		}
	}
	
	option = {
		name = sm_smsr.1.b
		ai_chance = { factor = 35 }
		custom_tooltip = sm_smsr.elections.tooltips
		add_republican_tradition = 0.1
		hidden_effect = {
			if = {
				limit = {
					faction_influence = {
						faction = mal_reformist_faction
						influence = 45
					}
					faction_influence = {
						faction = mal_democrat_faction
						influence = 10
					}
				}
				country_event = {
					id = sm_smsr.2
					days = 3
				}
			}
			if = {
				limit = {
					faction_influence = {
						faction = mal_democrat_faction
						influence = 45
					}
					faction_influence = {
						faction = mal_reformist_faction
						influence = 10
					}
				}
				country_event = {
					id = sm_smsr.2
					days = 3
				}
			}
			if = {
				limit = {
					OR = {
						faction_influence = {
							faction = mal_reformist_faction
							influence = 50
						}
						faction_influence = {
							faction = mal_democrat_faction
							influence = 50
						}
					}
				}
				country_event = {
					id = sm_smsr.2
					days = 3
				}
			}
		}
	}
}

country_event = {
	id = sm_smsr.6

	desc = sm_smsr.6.d
	picture = REVOLUTION_eventPicture
	
	major = yes
	
	fire_only_once = yes # ?
	
	trigger = {

		is_subject = no
		is_at_war = no
		NOT = { stability = 2 }
		faction_in_power = mal_totalitarian_faction

		}

		}

	mean_time_to_happen = {

		modifier = {
			factor = 0.5
			NOT = { stability = 0 }
		}
		
		modifier = {
			factor = 0.75
			NOT = { republican_tradition = 0.5 }
		}
		
		modifier = {
			factor = 0.5
			NOT = { republican_tradition = 0.25 }
		}
		
		modifier = {
			factor = 0.75
			faction_influence = {
				faction = mal_reformist_faction
				influence = 20
			}
		}
		
		modifier = {
			factor = 0.75
			faction_influence = {
				faction = mal_reformist_faction
				influence = 24
			}
		}
	}
	
	option = {
		name = sm_smsr.6.a
		ai_chance = { factor = 90 }
		release = DMF
		add_faction_influence = {
			faction = mal_democrat_faction
			influence = -10
		}
		hidden_effect = {
			random_owned_province = {
				limit = {
					units_in_province = 10
				}
				#kill_units = {
				#	who = owner
				#	amount = 20
				#}
			}
			add_country_modifier = {
				name = civil_war_nation
				duration = -1
			}
			DMF = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.7
						cavalry = 0.3
					}
				}
				add_country_modifier = {
					name = civil_war_nation
					duration = -1
				}
				change_government = socialist_republic
				add_stability = 1
				set_country_flag = sm_will_get_troops_in_civil_war
				every_owned_province = {
					limit = { units_in_province = 1 }
					kill_units = {
						who = enemy
						amount = 20
					}
				}
			}
		}
		declare_war_with_cb = {
			who = DMF
			casus_belli = cb_civil_war
		}
	}
}
