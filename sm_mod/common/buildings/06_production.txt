# Мануфактуры
primitive_manufactories = {
	cost = 150
	time = 12
	trigger = {
		FROM = {
			is_tribal = no
	#		NOT = { has_country_modifier = sm_uncivilized_nation }
		}
	}
	modifier = {
		local_production_efficiency = 0.25
	}
	
	allow_in_gold_provinces = no

	ai_will_do = {
		factor = 100
		modifier = {
			factor = 0
			owner = {
				is_at_war = yes
			}
		}
		modifier = {
			factor = 0
			NOT = { base_production = 3 }
		}		
		modifier = {
			factor = 0.1
			NOT = { base_production = 5 }
		}
		modifier = {
			factor = 1.5
			base_production = 7
		}		
		modifier = {
			factor = 1.5
			base_production = 9
		}		
		#modifier = {
		#	factor = 0.1
		#	is_overseas = yes
		#}
	}
}
